name: Air Quality Daily Report

on:
  schedule:
    # ตั้งค่าเวลา UTC (เวลาไทยลบด้วย 7 ชม.)
    # 07:00 TH = 00:00 UTC
    # 12:00 TH = 05:00 UTC
    # 15:00 TH = 08:00 UTC
    # 17:00 TH = 10:00 UTC
    - cron: '0 0,5,8,10 * * *'
  workflow_dispatch: # ปุ่มกดรันเองเพื่อทดสอบ

jobs:
  report:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout repository
        uses: actions/checkout@v3

      - name: Set up Python
        uses: actions/setup-python@v4
        with:
          python-version: '3.9'

      - name: Install dependencies
        run: pip install -r requirements.txt

      # ขั้นตอนการรันสคริปต์บอท พร้อมระบุ Secret ทั้งหมดที่ต้องใช้
      - name: Run Report Script
        env:
          AIR4THAI_KEY: ${{ secrets.AIR4THAI_KEY }}
          GISTDA_API_KEY: ${{ secrets.GISTDA_API_KEY }}
          TMD_3HR_KEY: ${{ secrets.TMD_3HR_KEY }}
          TELEGRAM_BOT_TOKEN: ${{ secrets.TELEGRAM_BOT_TOKEN }}
          TELEGRAM_CHAT_IDS: ${{ secrets.TELEGRAM_CHAT_IDS }}
        run: python bot.py

      # เก็บประวัติข้อมูลจุดความร้อนเพื่อใช้เปรียบเทียบแนวโน้มในครั้งถัดไป
      - name: Commit and Push History (to keep track of trends)
        run: |
          git config --global user.name "github-actions[bot]"
          git config --global user.email "github-actions[bot]@users.noreply.github.com"
          git add hotspot_history.json || true
          git commit -m "Update hotspot history" || true
          git push || true
